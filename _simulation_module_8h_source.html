204 //! CHECK_DATA is used for the unforeseen situation
205 #define CHECK_DATA(moduleID, expression, desc) if ((expression)) \
206  throw ModelException(moduleID, "CheckInputData", string(desc))
207 //! CHECK_POINTER is used for 1D or 2D raster and other pointer of data
208 #define CHECK_POINTER(moduleID, param) if (nullptr == (param)) \
209  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST NOT be NULL!"))
210 //! CHECK_POSITIVE is used for single value that must be positive
211 #define CHECK_POSITIVE(moduleID, param) if ((param) <= 0) \
212  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST be positive!"))
213 //! CHECK_NONNEGATIVE is used for single value that must be greater or equal than zero
214 #define CHECK_NONNEGATIVE(moduleID, param) if ((param) < 0) \
215  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST be greater or equal than zero!"))
216 //! CHECK_NEGATIVE is used for single value that must be negative
217 #define CHECK_NEGATIVE(moduleID, param) if ((param) >= 0) \
218  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST be negative!"))
219 //! CHECK_ZERO is used for single value that must not be ZERO
220 #define CHECK_ZERO(moduleID, param) if ((param) == 0 || FloatEqual(CVT_FLT(param), 0.f)) \
221  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST NOT be zero!"))
222 //! CHECK_NODATA is used for single value that must not be NODATA_VALUE
223 #define CHECK_NODATA(moduleID, param) if ((param) == NODATA_VALUE || FloatEqual(CVT_FLT(param), NODATA_VALUE)) \
224  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST NOT be NODATA_VALUE!"))
