1045 template <typename T, typename MASK_T>
1046 clsRasterData<T, MASK_T>::clsRasterData(vector<string>& filenames,
1047  const bool calc_pos /* = true */,
1048  clsRasterData<MASK_T>* mask /* = nullptr */,
1049  const bool use_mask_ext /* = true */,
1050  T default_value /* = static_cast<T>(NODATA_VALUE) */) {
1051  InitializeRasterClass();
1052  if (filenames.empty()) {
1053  // if filenames is empty
1054  StatusNoUnitTest("The filenames MUST have at least one raster file path!");
1055  return;
1056  }
1057  if (!CheckRasterFilesExist(filenames)) {
1058  // if not all filenames are presented
1059  StatusNoUnitTest("Please make sure all file path existed!");
1060  return;
1061  }
1062  if (filenames.size() == 1) {
1063  // if filenames has only one file
1064  if (!ConstructFromSingleFile(filenames[0], calc_pos, mask,
1065  use_mask_ext, default_value)) {
1066  return;
1067  }
1068  } else {
1069  // construct from multi-layers file
1070  n_lyrs_ = CVT_INT(filenames.size());
1071  // 1. firstly, take the first layer as the main input, to calculate position index or
1072  // extract by mask if stated.
1073  if (!ConstructFromSingleFile(filenames[0], calc_pos, mask,
1074  use_mask_ext, default_value)) {
1075  return;
1076  }
1077  // 2. then, change the core file name and file path template which format is: `<file dir>/CoreName_%d.<suffix>`
1078  core_name_ = SplitString(core_name_, '_').at(0);
1079  full_path_ = GetPathFromFullName(filenames[0]) + SEP + core_name_ + "_%d." + GetSuffix(filenames[0]);
1080  // So, to get a given layer's filepath, please use the following code. Definitely, maximum 99 layers is supported now.
1081  // string layerFilepath = m_filePathName.replace(m_filePathName.find_last_of("%d") - 1, 2, ValueToString(1));
1082  // 3. initialize raster_2d_ and read the other layers according to position data if stated,
1083  // or just read by row and col
1084  Initialize2DArray(n_cells_, n_lyrs_, raster_2d_, no_data_value_);
1085 #pragma omp parallel for
1086  for (int i = 0; i < n_cells_; i++) {
1087  raster_2d_[i][0] = raster_[i];
1088  }
1089  Release1DArray(raster_);
1090  // take the first layer as mask, and use_mask_ext is true, and no need to calculate position data
1091  for (int fileidx = 1; fileidx < CVT_INT(filenames.size()); fileidx++) {
1092  map<string, double> tmpheader;
1093  T* tmplyrdata = nullptr;
1094  string curfilename = filenames[fileidx];
1095  if (StringMatch(GetUpper(GetSuffix(curfilename)), string(ASCIIExtension))) {
1096  ReadAscFile(curfilename, &tmpheader, &tmplyrdata);
1097  } else {
1098 #ifdef USE_GDAL
1099  ReadRasterFileByGdal(curfilename, &tmpheader, &tmplyrdata, &srs_);
1100 #else
1101  cout << "Warning: Only ASC format is supported without GDAL!" << endl;
1102  return;
1103 #endif
1104  }
1105  if (calc_pos_) {
1106 #pragma omp parallel for
1107  for (int i = 0; i < n_cells_; ++i) {
1108  int tmp_row = raster_pos_data_[i][0];
1109  int tmp_col = raster_pos_data_[i][1];
1110  AddOtherLayerRasterData(tmp_row, tmp_col, i, fileidx, tmpheader, tmplyrdata);
1111  }
1112  } else {
1113 #pragma omp parallel for
1114  for (int i = 0; i < CVT_INT(headers_.at(HEADER_RS_NROWS)); i++) {
1115  for (int j = 0; j < CVT_INT(headers_.at(HEADER_RS_NCOLS)); j++) {
1116  AddOtherLayerRasterData(i, j, i * CVT_INT(headers_.at(HEADER_RS_NCOLS)) + j,
1117  fileidx, tmpheader, tmplyrdata);
1118  }
1119  }
1120  }
1121  Release1DArray(tmplyrdata);
1122  }
1123  is_2draster = true;
1124  headers_.at(HEADER_RS_LAYERS) = n_lyrs_; // repair layers count in headers
1125  }
1126 }
