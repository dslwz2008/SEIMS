86  /*!
87  * \brief Read 2D array data and insert to m_2DArrayMap
88  *
89  * The matrix format is as follows:\n
90  * 5 (Row number) \n
91  * RowIdx 0 1 2 3 4 (ColIdx) \n
92  * 0 1 9. \n
93  * 1 2 8. 1. \n
94  * 2 2 5. 2. \n
95  * 3 1 2. \n
96  * 4 4 2. 5. 1. 8. \n
97  *
98  * i.e., the first element in each row is the valid number of the current row.
99  *
100  * \param[in] remote_filename data file name
101  * \param[out] rows first dimension of the 2D Array, i.e., Rows
102  * \param[out] cols second dimension of the 2D Array, i.e., Cols. If each col are different, set cols to 1.
103  * \param[out] data returned data
104  */
105  virtual void Read2DArrayData(const string& remote_filename, int& rows, int& cols, float**& data) = 0;
