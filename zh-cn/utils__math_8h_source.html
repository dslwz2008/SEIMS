164 template <typename T>
165 void BasicStatistics(const T* values, const int num, double** derivedvalues,
166  T exclude /* = CVT_TYP(NODATA_VALUE) */) {
167  double* tmpstats = new double[6];
168  double maxv = MISSINGFLOAT;
169  double minv = MAXIMUMFLOAT;
170  int validnum = 0;
171  double sumv = 0.;
172  double std = 0.;
173  for (int i = 0; i < num; i++) {
174  if (FloatEqual(values[i], exclude)) continue;
175  if (maxv < values[i]) maxv = values[i];
176  if (minv > values[i]) minv = values[i];
177  validnum += 1;
178  sumv += values[i];
179  }
180  tmpstats[0] = CVT_DBL(validnum);
181  double mean = sumv / tmpstats[0];
182 #pragma omp parallel for reduction(+:std)
183  for (int i = 0; i < num; i++) {
184  if (!FloatEqual(values[i], exclude)) {
185  std += (values[i] - mean) * (values[i] - mean);
186  }
187  }
188  std = sqrt(std / tmpstats[0]);
189  tmpstats[1] = mean;
190  tmpstats[2] = maxv;
191  tmpstats[3] = minv;
192  tmpstats[4] = std;
193  tmpstats[5] = maxv - minv;
194  *derivedvalues = tmpstats;
195 }

197 template <typename T>
198 void BasicStatistics(const T*const * values, const int num, const int lyrs,
199  double*** derivedvalues, T exclude /* = CVT_TYP(NODATA_VALUE) */) {
200  double** tmpstats = new double *[6];
201  for (int i = 0; i < 6; i++) {
202  tmpstats[i] = new double[lyrs];
203  }
204  for (int j = 0; j < lyrs; j++) {
205  tmpstats[0][j] = 0.; /// valid number
206  tmpstats[1][j] = 0.; /// mean
207  tmpstats[2][j] = CVT_DBL(MISSINGFLOAT); /// maximum
208  tmpstats[3][j] = CVT_DBL(MAXIMUMFLOAT); /// minimum
209  tmpstats[4][j] = 0.; /// std
210  tmpstats[5][j] = 0.; /// range
211  }
212  double* sumv = nullptr;
213  utils_array::Initialize1DArray(lyrs, sumv, 0.);
214  for (int i = 0; i < num; i++) {
215  for (int j = 0; j < lyrs; j++) {
216  if (FloatEqual(values[i][j], exclude)) continue;
217  if (tmpstats[2][j] < values[i][j]) tmpstats[2][j] = values[i][j];
218  if (tmpstats[3][j] > values[i][j]) tmpstats[3][j] = values[i][j];
219  tmpstats[0][j] += 1;
220  sumv[j] += values[i][j];
221  }
222  }
223 
224  for (int j = 0; j < lyrs; j++) {
225  tmpstats[5][j] = tmpstats[2][j] - tmpstats[3][j];
226  tmpstats[1][j] = sumv[j] / tmpstats[0][j];
227  }
228  for (int j = 0; j < lyrs; j++) {
229  double tmpstd = 0;
230 #pragma omp parallel for reduction(+:tmpstd)
231  for (int i = 0; i < num; i++) {
232  if (!FloatEqual(values[i][j], exclude)) {
233  tmpstd += (values[i][j] - tmpstats[1][j]) * (values[i][j] - tmpstats[1][j]);
234  }
235  }
236  tmpstats[4][j] = tmpstd;
237  }
238  for (int j = 0; j < lyrs; j++) {
239  tmpstats[4][j] = sqrt(tmpstats[4][j] / tmpstats[0][j]);
240  }
241  utils_array::Release1DArray(sumv);
242  *derivedvalues = tmpstats;
243 }
