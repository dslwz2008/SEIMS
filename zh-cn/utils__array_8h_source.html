180 template <typename T, typename INI_T>
181 bool Initialize1DArray(const int row, T*& data, const INI_T init_value) {
182  if (nullptr != data) {
183  cout << "The input 1D array pointer is not nullptr, without initialized!" << endl;
184  return false;
185  }
186  data = new(nothrow)T[row];
187  if (nullptr == data) {
188  delete[] data;
189  cout << "Bad memory allocated during 1D array initialization!" << endl;
190  return false;
191  }
192  T init = static_cast<T>(init_value);
193 #pragma omp parallel for
194  for (int i = 0; i < row; i++) {
195  data[i] = init;
196  }
197  return true;
198 }

223 template <typename T, typename INI_T>
224 bool Initialize2DArray(const int row, const int col, T**& data,
225  const INI_T init_value) {
226  if (nullptr != data) {
227  cout << "The input 2D array pointer is not nullptr, without initialized!" << endl;
228  return false;
229  }
230  data = new(nothrow) T *[row];
231  if (nullptr == data) {
232  delete[] data;
233  cout << "Bad memory allocated during 2D array initialization!" << endl;
234  return false;
235  }
236  T init = static_cast<T>(init_value);
237  int bad_alloc = 0;
238 #pragma omp parallel for reduction(+:bad_alloc)
239  for (int i = 0; i < row; i++) {
240  data[i] = new(nothrow) T[col];
241  if (nullptr == data[i]) {
242  delete[] data[i];
243  bad_alloc++;
244  }
245  for (int j = 0; j < col; j++) {
246  data[i][j] = init;
247  }
248  }
249  if (bad_alloc > 0) {
250  cout << "Bad memory allocated during 2D array initialization!" << endl;
251  Release2DArray(row, data);
252  return false;
253  }
254  return true;
255 }

257 template <typename T, typename INI_T>
258 bool Initialize2DArray(const int row, const int col, T**& data,
259  INI_T** const init_data) {
260  if (nullptr != data) {
261  cout << "The input 2D array pointer is not nullptr, without initialized!" << endl;
262  return false;
263  }
264  data = new(nothrow)T *[row];
265  if (nullptr == data) {
266  delete[] data;
267  cout << "Bad memory allocated during 2D array initialization!" << endl;
268  return false;
269  }
270  int bad_alloc = 0;
271  int error_access = 0;
272 #pragma omp parallel for reduction(+:bad_alloc, error_access)
273  for (int i = 0; i < row; i++) {
274  data[i] = new(nothrow)T[col];
275  if (nullptr == data[i]) {
276  delete[] data[i];
277  bad_alloc++;
278  }
279  if (nullptr == init_data[i]) {
280  error_access++;
281  } else {
282  for (int j = 0; j < col; j++) {
283  data[i][j] = static_cast<T>(init_data[i][j]);
284  }
285  }
286  }
287  if (bad_alloc > 0) {
288  cout << "Bad memory allocated during 2D array initialization!" << endl;
289  Release2DArray(row, data);
290  return false;
291  }
292  if (error_access > 0) {
293  cout << "nullptr pointer existed in init_data during 2D array initialization!" << endl;
294  Release2DArray(row, data);
295  return false;
296  }
297  return true;
298 }

308 template <typename T>
309 void Release2DArray(const int row, T**& data) {
310  if (nullptr == data) {
311  return;
312  }
313 #pragma omp parallel for
314  for (int i = 0; i < row; i++) {
315  if (data[i] != nullptr) {
316  delete[] data[i];
317  data[i] = nullptr;
318  }
319  }
320  delete[] data;
321  data = nullptr;
322 }

326  va_list arg_ptr;
327  va_start(arg_ptr, data);
328  Release1DArray(data);
