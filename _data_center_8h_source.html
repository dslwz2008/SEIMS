83  /*!
84  * \brief Read 2D array data and insert to m_2DArrayMap
85  *
86  * The matrix format is as follows:\n
87  * 5 (Row number) \n
88  * RowIdx 0 1 2 3 4 (ColIdx) \n
89  * 0 1 9. \n
90  * 1 2 8. 1. \n
91  * 2 2 5. 2. \n
92  * 3 1 2. \n
93  * 4 4 2. 5. 1. 8. \n
94  *
95  * i.e., the first element in each row is the valid number of the current row.
96  *
97  * \param[in] remote_filename data file name
98  * \param[out] rows first dimension of the 2D Array, i.e., Rows
99  * \param[out] cols second dimension of the 2D Array, i.e., Cols. If each col are different, set cols to 1.
100  * \param[out] data returned data
101  */
102  virtual void Read2DArrayData(const string& remote_filename, int& rows, int& cols, float**& data) = 0;
