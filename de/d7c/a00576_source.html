45 int IUH_SED_OL::Execute() {
46  CheckInputData();
47  InitialOutputs();
48  // delete value of last time step
49  for (int i = 0; i < m_nSubbsns + 1; i++) {
50  m_sedtoCh[i] = 0.f;
51  }
52 
53 #pragma omp parallel for
54  for (int i = 0; i < m_nCells; i++) {
55  //forward one time step
56  for (int j = 0; j < m_cellFlowCols; j++) {
57  if (j != m_cellFlowCols - 1) {
58  m_cellSed[i][j] = m_cellSed[i][j + 1];
59  } else {
60  m_cellSed[i][j] = 0.f;
61  }
62  }
63 
64  if (m_sedYield[i] > 0.f) {
65  int min = CVT_INT(m_iuhCell[i][0]);
66  int max = CVT_INT(m_iuhCell[i][1]);
67  int col = 2;
68  for (int k = min; k <= max; k++) {
69  m_cellSed[i][k] += m_sedYield[i] * m_iuhCell[i][col];
70  col++;
71  }
72  }
73  }
74  // See https://github.com/lreis2415/SEIMS/issues/36 for more descriptions. By lj
75 #pragma omp parallel
76  {
77  float* tmp_sed2ch = new float[m_nSubbsns + 1];
78  for (int i = 0; i <= m_nSubbsns; i++) {
79  tmp_sed2ch[i] = 0.f;
80  }
81 #pragma omp for
82  for (int i = 0; i < m_nCells; i++) {
83  tmp_sed2ch[CVT_INT(m_subbsnID[i])] += m_cellSed[i][0];
84  m_olWtrEroSed[i] = m_cellSed[i][0];
85  }
86 #pragma omp critical
87  {
88  for (int i = 1; i <= m_nSubbsns; i++) {
89  m_sedtoCh[i] += tmp_sed2ch[i];
90  }
91  }
92  delete[] tmp_sed2ch;
93  tmp_sed2ch = nullptr;
94  } /* END of #pragma omp parallel */
95  for (int i = 1; i < m_nSubbsns + 1; i++) {
96  m_sedtoCh[0] += m_sedtoCh[i]; //get overland flow routing for entire watershed.
97  }
98  return 0;
99 }
